MACRO(build_test name)
    ADD_EXECUTABLE(${name} ${name}.cpp)
    IF (MINGW)
        TARGET_LINK_LIBRARIES(${name}
            gmock
            ${PROFILE_RT_LIBS}
            ${CLANG_LIBRARIES}
            ${REQ_LLVM_LIBRARIES}
            OCLintMetric
            OCLintAbstractRule
            OCLintCore
            OCLintRuleSet
        )
    ELSE()
        TARGET_LINK_LIBRARIES(${name}
            OCLintAbstractRule
            OCLintMetric
            OCLintHelper
            OCLintUtil
            OCLintCore
            OCLintRuleSet
            gmock
            ${PROFILE_RT_LIBS}
            ${CLANG_LIBRARIES}
            ${REQ_LLVM_LIBRARIES}
        )
    ENDIF()

    ADD_TEST(${name} ${EXECUTABLE_OUTPUT_PATH}/${name})
ENDMACRO(build_test)
